Location access failed: ${errorMessage}.

`; console.error('Geolocation error:', errorMessage, 'Code:', error.code); }, { timeout: 15000, // Increased timeout enableHighAccuracy: true, // High accuracy for better results maximumAge: 0 // Prevent cached position } ); // Optional: Watch position for updates (if needed) const watchId = navigator.geolocation.watchPosition( position => { const { latitude, longitude, accuracy } = position.coords; locationDisplay.textContent = `Latitude: ${latitude.toFixed(4)}, Longitude: ${longitude.toFixed(4)}, Accuracy: ${accuracy.toFixed(0)} meters`; locationError.textContent = 'None'; }, error => { console.error('WatchPosition error:', error.message, 'Code:', error.code); }, { timeout: 15000, enableHighAccuracy: true } ); // Clean up watchPosition after 30 seconds to avoid unnecessary resource use setTimeout(() => navigator.geolocation.clearWatch(watchId), 30000); } else { locationDisplay.textContent = 'Geolocation not supported'; locationError.textContent = 'Browser does not support geolocation'; promptDiv.innerHTML = '
